import js from js import document, FileReader, URL, Blob, Event, console from pyodide.ffi import create_proxy import pandas as pd from io import StringIO df1 = None df2 = None def update_status(msg, color="#444"): document.getElementById("status").innerText = msg document.getElementById("status").style.color = color console.log("[STATUS] " + msg) def check_ready(): btn = document.getElementById("mergeBtn") if df1 is not None and df2 is not None: btn.disabled = False update_status("Both files loaded – ready to merge", "#27ae60") else: btn.disabled = True update_status("Waiting for both files…") def parse_csv(event, target): global df1, df2 content = event.target.result try: df = pd.read_csv(StringIO(content)) if target == "df1": df1 = df else: df2 = df update_status(f"Loaded {target.upper()} ({len(df)} rows)", "#555") check_ready() except Exception as err: update_status(f"Error reading CSV: {err}", "crimson") console.error(err) def load_file(e, target): file = e.target.files[0] if not file: return name_el = document.getElementById("name1" if target == "df1" else "name2") name_el.innerText = file.name reader = FileReader.new() reader.onload = create_proxy(lambda ev: parse_csv(ev, target)) reader.readAsText(file) def merge_and_download(*args): global df1, df2 if df1 is None or df2 is None: return update_status("Merging…", "#1e90ff") try: # Check required columns for df, name in [(df1, "File 1"), (df2, "File 2")]: if 'A' not in df.columns or 'B' not in df.columns: raise ValueError(f"Missing column A or B in {name}") # Normalize keys (case + whitespace insensitive) df1_norm = df1.assign( __A_norm = df1['A'].astype(str).str.strip().str.lower(), __B_norm = df1['B'].astype(str).str.strip().str.lower() ) df2_norm = df2.assign( __A_norm = df2['A'].astype(str).str.strip().str.lower(), __B_norm = df2['B'].astype(str).str.strip().str.lower() ) # Find the value column in File 2 (first one that's not A or B) value_cols = [c for c in df2_norm.columns if c not in ['A', 'B', '__A_norm', '__B_norm']] if not value_cols: raise ValueError("No value column found in File 2 (only A and B present?)") value_col = value_cols[0] # Create lookup dictionary lookup = df2_norm.set_index(['__A_norm', '__B_norm'])[value_col].to_dict() # Add column C df1_norm['C'] = df1_norm.apply( lambda row: lookup.get((row['__A_norm'], row['__B_norm']), ""), axis=1 ) # Clean up temporary columns result = df1_norm.drop(columns=['__A_norm', '__B_norm']) # Generate CSV and download csv_content = result.to_csv(index=False) blob = Blob.new([csv_content], {"type": "text/csv"}) url = URL.createObjectURL(blob) a = document.createElement("a") a.href = url a.download = "merged_result.csv" a.click() URL.revokeObjectURL(url) update_status("Done! File downloaded as merged_result.csv", "#27ae60") except Exception as e: update_status(f"Error during merge: {e}", "crimson") console.error(e) # ── Event setup ──────────────────────────────────────────────── def setup_dropzone(drop_id, file_id): drop = document.getElementById(drop_id) file_input = document.getElementById(file_id) drop.addEventListener("click", lambda: file_input.click()) drop.addEventListener("dragover", lambda e: (e.preventDefault(), drop.classList.add("hover"))) drop.addEventListener("dragleave", lambda e: drop.classList.remove("hover")) drop.addEventListener("drop", lambda e: ( e.preventDefault(), drop.classList.remove("hover"), file_input.files = e.dataTransfer.files, file_input.dispatchEvent(Event.new("change")) )) setup_dropzone("drop1", "file1") setup_dropzone("drop2", "file2") document.getElementById("file1").addEventListener("change", create_proxy(lambda e: load_file(e, "df1"))) document.getElementById("file2").addEventListener("change", create_proxy(lambda e: load_file(e, "df2"))) document.getElementById("mergeBtn").addEventListener("click", create_proxy(merge_and_download)) update_status("Ready – waiting for files (pandas loading in background)", "#3498db")
