import js from js import document, FileReader, URL, Blob, Event from pyodide.ffi import create_proxy import pandas as pd from io import StringIO # Globals df1 = None df2 = None file1_name = "" file2_name = "" selected_value_col = "" def normalize_key(a, b): return (str(a).strip().lower(), str(b).strip().lower()) def update_status(msg, color="#444"): document.getElementById("status").innerText = msg document.getElementById("status").style.color = color def check_ready(): btn = document.getElementById("mergeBtn") col_select = document.getElementById("valueColumn") has_files = df1 is not None and df2 is not None has_col = col_select.value.strip() != "" btn.disabled = not (has_files and has_col) if has_files and has_col: update_status("Ready to merge!", "#27ae60") elif has_files: update_status("Please select a value column from File 2", "#e67e22") else: update_status("Waiting for both files…", "#444") def populate_column_dropdown(): select = document.getElementById("valueColumn") select.innerHTML = '' if df2 is None: return # Exclude A and B from suggestions (but still show them in list for reference) options = [] for col in df2.columns: opt = document.createElement("option") opt.value = col opt.text = col if col not in ["A", "B"]: options.append(opt) # prefer non-key columns first else: select.appendChild(opt) # Add preferred ones after keys for opt in options: select.appendChild(opt) # Show section document.getElementById("columnSelects").classList.add("show") def on_value_column_change(event): global selected_value_col selected_value_col = event.target.value.strip() check_ready() def load_file1(e): global df1, file1_name file = e.target.files[0] if not file: return file1_name = file.name document.getElementById("name1").innerText = file1_name reader = FileReader.new() reader.onload = create_proxy(lambda ev: parse_csv(ev, "df1")) reader.readAsText(file) def load_file2(e): global df2, file2_name file = e.target.files[0] if not file: return file2_name = file.name document.getElementById("name2").innerText = file2_name reader = FileReader.new() reader.onload = create_proxy(lambda ev: parse_csv(ev, "df2")) reader.readAsText(file) def parse_csv(event, target): global df1, df2 content = event.target.result try: df = pd.read_csv(StringIO(content)) if target == "df1": df1 = df update_status(f"File 1 loaded ({len(df)} rows)", "#555") else: df2 = df update_status(f"File 2 loaded ({len(df)} rows)", "#555") populate_column_dropdown() check_ready() except Exception as err: update_status(f"CSV parse error: {err}", "crimson") def merge_and_download(*args): global df1, df2, selected_value_col if df1 is None or df2 is None or not selected_value_col: return update_status("Merging…", "#1e90ff") try: if 'A' not in df1.columns or 'B' not in df1.columns: raise ValueError("File 1 missing column 'A' or 'B'") if 'A' not in df2.columns or 'B' not in df2.columns: raise ValueError("File 2 missing column 'A' or 'B'") if selected_value_col not in df2.columns: raise ValueError(f"Selected column '{selected_value_col}' not found in File 2") # Normalize keys df1_norm = df1.assign( __A_norm = df1['A'].astype(str).str.strip().str.lower(), __B_norm = df1['B'].astype(str).str.strip().str.lower() ) df2_norm = df2.assign( __A_norm = df2['A'].astype(str).str.strip().str.lower(), __B_norm = df2['B'].astype(str).str.strip().str.lower() ) # Build lookup lookup = df2_norm.set_index(['__A_norm', '__B_norm'])[selected_value_col].to_dict() # Add C df1_norm['C'] = df1_norm.apply( lambda row: lookup.get((row['__A_norm'], row['__B_norm']), ""), axis=1 ) result_df = df1_norm.drop(columns=['__A_norm', '__B_norm']) # CSV → download csv_string = result_df.to_csv(index=False) blob = Blob.new([csv_string], {"type": "text/csv"}) url = URL.createObjectURL(blob) a = document.createElement("a") a.href = url a.download = "merged_result.csv" a.click() URL.revokeObjectURL(url) update_status(f"Done! Downloaded merged_result.csv ({len(result_df)} rows)", "#27ae60") except Exception as err: update_status(f"Merge error: {err}", "crimson") print(err) # ── Event wiring ──────────────────────────────────────────── area1 = document.getElementById("area1") area1.addEventListener("click", lambda: document.getElementById("file1").click()) document.getElementById("file1").addEventListener("change", create_proxy(load_file1)) area2 = document.getElementById("area2") area2.addEventListener("click", lambda: document.getElementById("file2").click()) document.getElementById("file2").addEventListener("change", create_proxy(load_file2)) # Drag & drop for area, fid in [(area1, "file1"), (area2, "file2")]: area.addEventListener("dragover", lambda e: (e.preventDefault(), area.classList.add("hover"))) area.addEventListener("dragleave", lambda e: area.classList.remove("hover")) area.addEventListener("drop", lambda e, f=fid: ( e.preventDefault(), area.classList.remove("hover"), document.getElementById(f).files = e.dataTransfer.files, document.getElementById(f).dispatchEvent(Event.new("change")) )) document.getElementById("mergeBtn").addEventListener("click", create_proxy(merge_and_download)) document.getElementById("valueColumn").addEventListener("change", create_proxy(on_value_column_change))
